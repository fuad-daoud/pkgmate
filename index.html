pkgmate APT Repository

Installation

curl -fsSL https://fuad-daoud.github.io/pkgmate/public.key | sudo gpg --dearmor -o /usr/share/keyrings/pkgmate-archive-keyring.gpg
echo "deb [arch=amd64,arm64 signed-by=/usr/share/keyrings/pkgmate-archive-keyring.gpg] https://fuad-daoud.github.io/pkgmate stable main" | sudo tee /etc/apt/sources.list.d/pkgmate.list
sudo apt update && sudo apt install pkgmate
